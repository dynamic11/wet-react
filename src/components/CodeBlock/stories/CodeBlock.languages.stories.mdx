export const JavascriptTemplate = (args) => (
  <CodeBlock {...args}>
    {`function createStyleObject(classNames, style) {
    return classNames.reduce((styleObject, className) => {
      return { ...styleObject, ...style[className] };
    }, {});
  }`}
  </CodeBlock>
);

# CodeBlock

## Common Language Options

We have configured `React Syntax Highlighter` to use `prism.js` as the core formatter therefore the supported langauges can be found [here](https://github.com/react-syntax-highlighter/react-syntax-highlighter/blob/HEAD/AVAILABLE_LANGUAGES_PRISM.MD).

### Javascript

Set `language` to `javascript`

<Canvas>
  <Story name="Javascript" args={{ language: 'javascript' }}>
    {JavascriptTemplate.bind({})}
  </Story>
</Canvas>

### HTML

Set `language` to `html`

<Canvas>
  <Story name="HTML" args={{ language: 'markup' }}>
    {HTMLTemplate.bind({})}
  </Story>
</Canvas>

### CSS

Set `language` to `css`

<Canvas>
  <Story name="CSS" args={{ language: 'css' }}>
    {CSSTemplate.bind({})}
  </Story>
</Canvas>

### Java

Set `language` to `java`

<Canvas>
  <Story name="Java" args={{ language: 'java' }}>
    {JavaTemplate.bind({})}
  </Story>
</Canvas>
